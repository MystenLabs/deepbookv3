```rust
/// Ensures `s` has the `0x` prefix. Returns owned String.
/// Input "abc123" → "0xabc123". Input "0xabc123" → "0xabc123".
pub fn normalize_asset_id(s: &str) -> String {
    if s.starts_with("0x") {
        s.to_string()
    } else {
        format!("0x{}", s)
    }
}

/// Removes the `0x` prefix from `s` if present. Returns owned String.
/// Input "0xabc123" → "abc123". Input "abc123" → "abc123".
pub fn strip_asset_prefix(s: &str) -> String {
    if let Some(stripped) = s.strip_prefix("0x") {
        stripped.to_string()
    } else {
        s.to_string()
    }
}
```

No other functions. No imports needed (pure string manipulation).
