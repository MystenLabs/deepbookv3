```typescript
import type {
  DeepBookPool,
  TradeRecord,
  TradesParams,
  OrderbookResponse,
  OhclvResponse,
  OhclvParams,
  PortfolioQueryResult,
  OrderRecord,
  OrdersParams,
  TickerEntry,
  SummaryEntry,
  AssetEntry,
  OrderUpdateRecord,
  OrderUpdatesParams,
  DepositedAsset,
  PointsEntry,
  StatusResponse,
} from "./types.js";

/**
 * HTTP client for the DeepBook v3 indexer REST API.
 *
 * @example
 * ```ts
 * const client = new DeepBookIndexerClient("https://deepbook-indexer.example.com");
 * const pools = await client.getPools();
 * console.log(pools[0].pool_name); // "DEEP_SUI"
 * ```
 */
export class DeepBookIndexerClient {
  private readonly baseUrl: string;

  /**
   * @param baseUrl - Base URL of the indexer (no trailing slash).
   *   Defaults to "http://localhost:8080".
   */
  constructor(baseUrl = "http://localhost:8080") {
    this.baseUrl = baseUrl.replace(/\/$/, "");
  }

  private async get<T>(path: string): Promise<T> {
    const res = await fetch(`${this.baseUrl}${path}`);
    if (!res.ok) {
      throw new Error(`DeepBook indexer error ${res.status}: ${path}`);
    }
    return res.json() as Promise<T>;
  }

  private buildQuery(params: Record<string, string | number | undefined>): string {
    const entries = Object.entries(params).filter(([, v]) => v !== undefined);
    if (entries.length === 0) return "";
    return "?" + entries.map(([k, v]) => `${k}=${encodeURIComponent(String(v))}`).join("&");
  }

  /**
   * Returns all registered trading pools.
   *
   * @example
   * ```ts
   * const pools = await client.getPools();
   * // pools[0] → { pool_id: "0x...", pool_name: "DEEP_SUI", base_asset: "DEEP", ... }
   * ```
   */
  async getPools(): Promise<DeepBookPool[]> {
    return this.get<DeepBookPool[]>("/get_pools");
  }

  /**
   * Returns trade records for a pool, optionally filtered by time range and balance manager.
   *
   * @param poolName - Pool identifier, e.g. "DEEP_SUI".
   * @param params - Optional query filters.
   *
   * @example
   * ```ts
   * const trades = await client.getTrades("DEEP_SUI", { limit: 10 });
   * // trades[0] → { trade_id: "...", price: 0.05, base_quantity: 100, ... }
   * ```
   */
  async getTrades(poolName: string, params: TradesParams = {}): Promise<TradeRecord[]> {
    const q = this.buildQuery(params as Record<string, string | number | undefined>);
    return this.get<TradeRecord[]>(`/trades/${encodeURIComponent(poolName)}${q}`);
  }

  /**
   * Returns the current level-2 order book for a pool.
   *
   * @param poolName - Pool identifier, e.g. "DEEP_SUI".
   *
   * @example
   * ```ts
   * const book = await client.getOrderbook("DEEP_SUI");
   * // book.bids[0] → ["0.0500", "1000"]  // [price, quantity]
   * ```
   */
  async getOrderbook(poolName: string): Promise<OrderbookResponse> {
    return this.get<OrderbookResponse>(`/orderbook/${encodeURIComponent(poolName)}`);
  }

  /**
   * Returns OHCLV (open/high/close/low/volume) candlestick data for a pool.
   *
   * @param poolName - Pool identifier, e.g. "DEEP_SUI".
   * @param params - Time range and candle interval.
   *
   * @example
   * ```ts
   * const result = await client.getOhclv("DEEP_SUI", { interval_in_ms: 3600000 });
   * // result.candles[0] → { open: 0.05, high: 0.06, close: 0.055, low: 0.049, volume: 5000, timestamp: 1700000000 }
   * ```
   */
  async getOhclv(poolName: string, params: OhclvParams = {}): Promise<OhclvResponse> {
    const q = this.buildQuery(params as Record<string, string | number | undefined>);
    return this.get<OhclvResponse>(`/ohclv/${encodeURIComponent(poolName)}${q}`);
  }

  /**
   * Returns the aggregated portfolio for a wallet address.
   *
   * @param walletAddress - Sui wallet address (0x-prefixed).
   *
   * @example
   * ```ts
   * const portfolio = await client.getPortfolio("0xabc...");
   * // portfolio.balances[0] → { asset: "DEEP", available: "100.0", locked: "10.0" }
   * ```
   */
  async getPortfolio(walletAddress: string): Promise<PortfolioQueryResult> {
    return this.get<PortfolioQueryResult>(`/portfolio/${encodeURIComponent(walletAddress)}`);
  }

  /**
   * Returns orders for a balance manager in a pool.
   *
   * @param poolName - Pool identifier, e.g. "DEEP_SUI".
   * @param balanceManagerId - Balance manager object ID (0x-prefixed).
   * @param params - Optional filters (limit, time range, status).
   *
   * @example
   * ```ts
   * const orders = await client.getOrders("DEEP_SUI", "0xbm...", { status: "filled" });
   * // orders[0] → { order_id: "...", price: 0.05, quantity: 100, status: "filled", ... }
   * ```
   */
  async getOrders(
    poolName: string,
    balanceManagerId: string,
    params: OrdersParams = {}
  ): Promise<OrderRecord[]> {
    const q = this.buildQuery(params as Record<string, string | number | undefined>);
    return this.get<OrderRecord[]>(
      `/orders/${encodeURIComponent(poolName)}/${encodeURIComponent(balanceManagerId)}${q}`
    );
  }

  /**
   * Returns 24h ticker statistics keyed by pool name.
   *
   * @example
   * ```ts
   * const tickers = await client.getTicker();
   * // tickers["DEEP_SUI"] → { last_price: "0.05", base_volume: "100000", ... }
   * ```
   */
  async getTicker(): Promise<Record<string, TickerEntry>> {
    return this.get<Record<string, TickerEntry>>("/ticker");
  }

  /**
   * Returns market summary entries for all trading pairs.
   *
   * @example
   * ```ts
   * const summary = await client.getSummary();
   * // summary[0] → { trading_pairs: "DEEP_SUI", last_price: "0.05", ... }
   * ```
   */
  async getSummary(): Promise<SummaryEntry[]> {
    return this.get<SummaryEntry[]>("/summary");
  }

  /**
   * Returns asset metadata keyed by asset symbol.
   *
   * @example
   * ```ts
   * const assets = await client.getAssets();
   * // assets["DEEP"] → { name: "DEEP", can_deposit: true, maker_fee: "0.001", ... }
   * ```
   */
  async getAssets(): Promise<Record<string, AssetEntry>> {
    return this.get<Record<string, AssetEntry>>("/assets");
  }

  /**
   * Returns order lifecycle events (placed, filled, cancelled) for a balance manager.
   *
   * @param poolName - Pool identifier, e.g. "DEEP_SUI".
   * @param balanceManagerId - Balance manager object ID.
   * @param params - Optional filters (limit, time range).
   *
   * @example
   * ```ts
   * const updates = await client.getOrderUpdates("DEEP_SUI", "0xbm...", { limit: 50 });
   * // updates[0] → { order_id: "...", event_type: "filled", quantity: 100, ... }
   * ```
   */
  async getOrderUpdates(
    poolName: string,
    balanceManagerId: string,
    params: OrderUpdatesParams = {}
  ): Promise<OrderUpdateRecord[]> {
    const q = this.buildQuery(params as Record<string, string | number | undefined>);
    return this.get<OrderUpdateRecord[]>(
      `/order_updates/${encodeURIComponent(poolName)}/${encodeURIComponent(balanceManagerId)}${q}`
    );
  }

  /**
   * Returns assets deposited into all balance managers for a wallet address.
   *
   * @param walletAddress - Sui wallet address (0x-prefixed).
   *
   * @example
   * ```ts
   * const assets = await client.getDepositedAssets("0xabc...");
   * // assets[0] → { balance_manager_id: "0xbm...", asset: "DEEP", amount: "500.0" }
   * ```
   */
  async getDepositedAssets(walletAddress: string): Promise<DepositedAsset[]> {
    return this.get<DepositedAsset[]>(`/deposited_assets/${encodeURIComponent(walletAddress)}`);
  }

  /**
   * Returns net deposit amounts per asset for a wallet address.
   * Keys are asset symbols; values are net amounts (deposits minus withdrawals).
   *
   * @param walletAddress - Sui wallet address (0x-prefixed).
   *
   * @example
   * ```ts
   * const net = await client.getNetDeposits("0xabc...");
   * // net["DEEP"] → 250  (net 250 DEEP deposited)
   * ```
   */
  async getNetDeposits(walletAddress: string): Promise<Record<string, number>> {
    return this.get<Record<string, number>>(
      `/get_net_deposits/${encodeURIComponent(walletAddress)}`
    );
  }

  /**
   * Returns trading points earned by a wallet address across all pools.
   *
   * @param walletAddress - Sui wallet address (0x-prefixed).
   *
   * @example
   * ```ts
   * const points = await client.getPoints("0xabc...");
   * // points[0] → { pool_name: "DEEP_SUI", maker_volume: "10000", points: "50" }
   * ```
   */
  async getPoints(walletAddress: string): Promise<PointsEntry[]> {
    return this.get<PointsEntry[]>(`/get_points/${encodeURIComponent(walletAddress)}`);
  }

  /**
   * Returns server health and current sync checkpoint.
   *
   * @example
   * ```ts
   * const status = await client.getStatus();
   * // status → { status: "ok", checkpoint: 12345678 }
   * ```
   */
  async getStatus(): Promise<StatusResponse> {
    return this.get<StatusResponse>("/status");
  }

  /**
   * Returns the total circulating supply of DEEP tokens.
   *
   * @example
   * ```ts
   * const supply = await client.getDeepSupply();
   * // supply → 10000000000
   * ```
   */
  async getDeepSupply(): Promise<number> {
    return this.get<number>("/deep_supply");
  }
}
```
